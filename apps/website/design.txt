function DocumentationHeader(item={APIItem}) { // Server
  switch (item.kind) {
    case APIItemKind.Class:
      return (
        <div>
          <Header text={...} />
          <Extends ... />
          <Implements ... />
        </div>
      );
    ...
  }
}

export function DocumentationBody(item={APIItem}) { // Server
  switch (item.kind) {
    case APIItemKind.Class:
      return (
        <div>
          <Properties item={item} />
          <Methods item={item} />
        </div>
      );
    ...
  }
}

export function Documentation(item: APIItem) { // Server
  return (
    <DocumentationHeader item={item} />
    
  )
}
